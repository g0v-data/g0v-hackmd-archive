#include <iostream>
using namespace std;

int main()
{
	int n=2,light,c;
	while(n<=1000)
	{
		light = 1;//燈打開 
		c = 2;// 可以整除的數字 
		while(c*c<=n)//比如81=9*9 ,9是質數 
		{
			if(n%c==0)//n被成功整除 
				light = 0;//把燈關掉 
			c++;//一直嘗試直到關掉 
		}
		if(light==1)
			cout << n << endl;//燈關不掉就是質數 
		n++;//while block的變數 
	}
	return 0;
}
